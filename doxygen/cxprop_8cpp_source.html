171 // in-place data reallocation only intended to work when only ion data has
172 // pointers to it and no one uses pointers to other prop data. If this does
173 // not hold, then segmentation violations are likely to occur.
174 // Note that the tml list is already ordered properly so ions are first.
175 // We can therefore call the mechanism pdata_update function (which normally
176 // contains pointers to ion variables) and the ion variables will exist
177 // in their new proper locations.
178 
179 // we do one type (for all threads) at a time. Sadly, we have to keep the
180 // original pool in existence til the new pool is complete. And we have to
181 // keep old ion pools til the end.

185 // extending to gui pointers and other well-known things that hold pointers
186 // the idea is that there are not *that* many.

188 static double* recalc_ptr(double* old) {
189  // is old in the op pool?
190  for (DoubleArrayPool* op = oldpools_[recalc_index_]; op; op = op->chain()) {
191  long ds = op->chain_size() * op->d2();
192  if (old >= op->pool() && old < (op->pool() + ds)) {
193  // if so then the value gives us the pointer in the new pool
194  long offset = old - op->pool();
195  offset %= op->d2();
196  DoubleArrayPool* np = dblpools_[recalc_index_];
197  long i = (long) (*old);
198  // if (i < 0 || i >= np->size()) abort();
199  assert(i >= 0 && i < np->size());
200  double* n = np->items()[i] + offset;
201  // printf("recalc_ptr old=%p new=%p value=%g\n", old, n, *n);
202  return n;
203  }
204  }

213  // what types are in use

221  for (NrnThreadMembList* tml = nt->tml; tml; tml = tml->next) {
222  ++types[tml->index];
223  }
224  }
225  // iterate over those types
226  for (int i = 0; i < n_memb_func; ++i)
227  if (types[i]) {
228  int ision = nrn_is_ion(i);
229  DoubleArrayPool* oldpool = dblpools_[i];
230  DoubleArrayPool* newpool = 0;
231  // create the pool with proper chain sizes
232  int ml_total_count = 0; // so we can know if we get them all
233  FOR_THREADS(nt) {
234  for (NrnThreadMembList* tml = nt->tml; tml; tml = tml->next)
235  if (i == tml->index) {
236  ml_total_count += tml->ml->nodecount;
237  if (!newpool) {
238  newpool = new DoubleArrayPool(tml->ml->nodecount, oldpool->d2());
239  } else {
240  newpool->grow(tml->ml->nodecount);
241  }
242  break;
243  }
244  }
245  // Any extras? Put them in a new last chain.
246  // actually ml_total_count cannot be 0.
247  int extra = oldpool->nget() - ml_total_count;
248  assert(extra >= 0 && newpool);
249  if (extra) {

251  }
252  newpool->free_all(); // items in pool data order
253  // reset ml->_data pointers to the new pool and copy the values
254  FOR_THREADS(nt) {

256  if (i == tml->index) {

258  for (int j = 0; j < ml->nodecount; ++j) {
259  double* data = ml->_data[j];
260  int ntget = newpool->ntget();
261  ml->_data[j] = newpool->alloc();
262  for (int k = 0; k < newpool->d2(); ++k) {
263  ml->_data[j][k] = data[k];
264  }
265  // store in old location enough info
266  // to construct a pointer to the new location
267  for (int k = 0; k < newpool->d2(); ++k) {
268  data[k] = double(ntget);
269  }
270  }
271  // update any Datum pointers to ion variable locations
272  void (*s)(Datum*) = memb_func[i]._update_ion_pointers;
273  if (s)
274  for (int j = 0; j < ml->nodecount; ++j) {
275  (*s)(ml->pdata[j]);
276  }
277  }
278  }

280  // efficiency. But there may be other old items,
281  // e.g MechanismStandard that are not part of the memb lists.
282  // We made the chain item for them above. Now, sadly, we
283  // need to iterate over the MechanismStandards to find the
284  // old pool data.

287  if (!mechstanlist_) {

289  mechstanlist_ = s->u.ctemplate->olist;

296  MechanismStandard* ms = (MechanismStandard*) (OBJ(q)->u.this_pointer);

305  }

307  // to construct a pointer to the new location
308  for (int k = 0; k < newpool->d2(); ++k) {
309  data[k] = double(ntget);
310  }
311  ++found;
312  }
313  }

315  // printf("%d extra %s, looked at %d, found %d\n", extra,
316  // memb_func[i].sym->name, looked_at, found);
317  assert(extra == found);
318  }

321  // let the gui and other things update
